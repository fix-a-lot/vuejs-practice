데이터 바인딩

v-model 디렉티브를 사용하면 사용자가 입력한 값과 화면에 보이는 값, 그리고 앱의 데이터가 동기화됩니다.

아래 입력란의 값을 바꿔보세요.


        createApp({
          data() {
            return {
              text: 'edit me!'
            }
          },
          methods: {
            fn1() {
              this.text = 'Hello world!';
            }
          }
        }).mount("#app");

{{ text }}

        vm.text = 'Hello world!';

        createApp({
          data() {

              obj: {
                a: 1,
                b: 2

            }
          },
          methods: {
            fn2() {
              console.log(this.obj); // Proxy {a: 37, b: 2}
              this.obj.a = this.obj.a + 1;
            }
          }
        }).mount("#app");
